599 // ---------------------------------------------------------------------------
600 // timeSpan
601 // ---------------------------------------------------------------------------
605 //
606 template < typename Iterator >
607 std::pair< double, double >
608 timeSpan( Iterator begin, Iterator end )
609 {
610  double tmin = 0., tmax = 0.;
611  if ( begin != end )
612  {
613  Iterator it = begin;
614  tmin = it->startTime();
615  tmax = it->endTime();
616  while( it != end )
617  {
618  tmin = std::min( tmin, it->startTime() );
619  tmax = std::max( tmax, it->endTime() );
620  ++it;
621  }
622  }
623  return std::make_pair( tmin, tmax );
624 }
