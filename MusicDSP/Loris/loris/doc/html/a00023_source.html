264 template< typename Container >
265 typename Container::iterator Distiller::distill( Container & partials )
266 {
267  // This can be done so much more easily and
268  // efficiently on a list than on other containers
269  // that it is worth copying the Partials to a
270  // list for distillation, and then transfering
271  // them back.
272  //
273  // See below for a specialization for the case
274  // of the Container being a list, so no copy
275  // is needed.
276  PartialList pl( partials.begin(), partials.end() );
277  PartialList::iterator it = distill_list( pl );
278 
279  // pl has distilled Partials at beginning, and
280  // unlabeled Partials at end:
281  typename Container::iterator beginUnlabeled =
282  std::copy( pl.begin(), it, partials.begin() );
283 
284  typename Container::iterator endUnlabeled =
285  std::copy( it, pl.end(), beginUnlabeled );
286 
287 
288  partials.erase( endUnlabeled, partials.end() );
289 
290  return beginUnlabeled;
291 }
