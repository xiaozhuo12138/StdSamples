233 template< typename Container >
234 typename Container::iterator
235 Collator::collate( Container & partials )
236 #else
237 inline
238 PartialList::iterator
239 Collator::collate( PartialList & partials )
240 #endif
241 {
242 #if ! defined(NO_TEMPLATE_MEMBERS)
243  typedef typename Container::iterator Iterator;
244 #else
245  typedef PartialList::iterator Iterator;
246 #endif
247 
248  // Partition the Partials into labeled and unlabeled,
249  // and collate the unlabeled ones and replace the
250  // unlabeled range.
251  // (This requires bidirectional iterator support.)
252  Iterator beginUnlabeled =
253  std::partition( partials.begin(), partials.end(),
254  std::not1( PartialUtils::isLabelEqual(0) ) );
255  // this used to be a stable partition, which
256  // is very much slower and seems unnecessary
257 
258  // cannot splice if this operation is to be generic
259  // with respect to container, have to copy:
260  PartialList collated( beginUnlabeled, partials.end() );
261  // collated.splice( collated.end(), beginUnlabeled, partials.end() );
262 
263  // determine the label for the first collated Partial:
264  Partial::label_type labelCollated = 1;
265  if ( partials.begin() != beginUnlabeled )
266  {
267  labelCollated =
268  1 + std::max_element( partials.begin(), beginUnlabeled,
269  PartialUtils::compareLabelLess() )->label();
270  }
271  if ( labelCollated < 1 )
272  {
273  labelCollated = 1;
274  }
275 
276  // collate unlabeled (zero-labeled) Partials:
277  collateAux( collated );
278 
279  // label the collated Partials:
280  for ( Iterator it = collated.begin(); it != collated.end(); ++it )
281  {
282  it->setLabel( labelCollated++ );
283  }
284 
285  // copy the collated Partials back into the source container
286  // after the range of labeled Partials
287  Iterator endCollated =
288  std::copy( collated.begin(), collated.end(), beginUnlabeled );
289 
290  // remove extra Partials from the end of the source container
291  if ( endCollated != partials.end() )
292  {
293  typename Iterator::difference_type numLabeled =
294  std::distance( partials.begin(), beginUnlabeled );
295 
296  partials.erase( endCollated, partials.end() );
297 
298  // restore beginUnlabeled:
299  beginUnlabeled = partials.begin();
300  std::advance( beginUnlabeled, numLabeled );
301  }
302  return beginUnlabeled;
303 }
