222  m_ffwdcoefs( ffwdbegin, ffwdend ),
223  m_fbackcoefs( fbackbegin, fbackend ),
224  m_delayline( std::max( ffwdend-ffwdbegin, fbackend-fbackbegin ) - 1, 0. ),
225  m_gain( gain )
226 {
227  if ( *fbackbegin == 0. )
228  {
229  Throw( InvalidObject,
230  "Tried to create a Filter with feeback coefficient at zero delay equal to 0.0" );
231  }
232 
233  // normalize the coefficients by 1/a[0], if a[0] is not equal to 1.0
234  // (already checked for a[0] == 0 above)
235  if ( *fbackbegin != 1. )
236  {
237  // scale all filter coefficients by a[0]:
238  std::transform( m_ffwdcoefs.begin(), m_ffwdcoefs.end(), m_ffwdcoefs.begin(),
239  std::bind2nd( std::divides<double>(), *fbackbegin ) );
240  std::transform( m_fbackcoefs.begin(), m_fbackcoefs.end(), m_fbackcoefs.begin(),
241  std::bind2nd( std::divides<double>(), *fbackbegin ) );
242  m_fbackcoefs[0] = 1.;
243  }
244 }
