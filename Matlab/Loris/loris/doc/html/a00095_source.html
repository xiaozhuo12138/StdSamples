350 void SpcFile::addPartials( PartialList::const_iterator begin_partials,
351  PartialList::const_iterator end_partials )
352 #endif
353 {
354  while ( begin_partials != end_partials )
355  {
356  // do not try to add unlabeled Partials, or
357  // Partials labeled beyond 256:
358  if ( 0 != begin_partials->label() && 257 > begin_partials->label() )
359  {
360  addPartial( *begin_partials );
361  }
362  ++begin_partials;
363  }
364 }
